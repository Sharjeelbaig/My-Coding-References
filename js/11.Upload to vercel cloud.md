# Upload to vercel cloud

```ts
import { put } from "@vercel/blob";
import dotenv from "dotenv";

dotenv.config();

function uploadFile(content, fileName) {
  return put(fileName, content, {
    access: 'public',
  });
}

const content = 'hello world';
const fileName = 'hello3.txt';

uploadFile(content, fileName)
  .then(({ url }) => {
    console.log('File URL:', url);
  })
  .catch((error) => {
    console.error('Error uploading file:', error);
  });
```

## Uploading image
```ts 
import { put } from "@vercel/blob";
import dotenv from "dotenv";

dotenv.config();

const content = 'data:image/jpeg;base64,/9j/4AAQSkZJRgABAQAAAQABAAD/2wBDAAMCAgMCAg...';

if (!process.env.BLOB_READ_WRITE_TOKEN) {
  console.error('Missing required env var BLOB_READ_WRITE_TOKEN');
  process.exit(1);
}

function decodeBase64DataUri(dataUri) {
  const [meta, base64Data] = dataUri.split(',');
  const match = meta.match(/^data:(.*);base64$/);
  const mimeType = match ? match[1] : 'application/octet-stream';
  const buffer = Buffer.from(base64Data, 'base64');
  return { buffer, mimeType };
}

function uploadBase64Image(dataUri, fileName) {
  const { buffer, mimeType } = decodeBase64DataUri(dataUri);
  return put(fileName, buffer, {
    access: 'public',
    contentType: mimeType,
  });
}

const fileName = process.env.BLOB_FILE_NAME || 'uploaded-image.jpg';

(async () => {
  try {
    const { url } = await uploadBase64Image(content, fileName);
    console.log('Upload successful:', url);
  } catch (error) {
    console.error('Upload failed:', error);
  }
})();
```
