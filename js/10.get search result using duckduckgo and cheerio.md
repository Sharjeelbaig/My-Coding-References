# Get search result using duckduckgo and cheerio
This example demonstrates how to perform a search on DuckDuckGo and extract the top 5 search result links using the Cheerio library in TypeScript.

```ts
import { load } from "cheerio";

export async function searchDuckDuckGo(query: string): Promise<string[]> {
  const url = `https://duckduckgo.com/html/?q=${encodeURIComponent(query)}`;
  const html = await fetch(url).then(r => r.text());

  const $ = load(html);
  const links = [];

  $(".result__title a").each((_, el) => {
    const link = $(el).attr("href");
    if (link && link.startsWith("http")) {
      links.push(link);
    }
  });

  return links.slice(0, 5);
}
```
